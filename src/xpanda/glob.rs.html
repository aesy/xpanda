18enum Token {
19    ExactStr(String),
20    // *
21    AnyStr,
22    // ?
23    AnyChar,
24    // [..]
25    AnyOf(Vec<Enclosed>),
26}

28#[derive(Debug, Clone)]
29enum Enclosed {
30    Char(char),
31    Range { from: char, to: char },
32}

40    pub fn compile(pattern: &str) -> Result<Self, Error> {
41        let mut reader = StrRead::new(pattern);
42        let mut tokens: Vec<Token> = Vec::new();
43
44        let push_char = |tokens: &mut Vec<Token>, c: char| {
45            if let Some(Token::ExactStr(s)) = tokens.last_mut() {
46                s.push(c);
47            } else {
48                tokens.push(Token::ExactStr(c.to_string()));
49            }
50        };
51
52        while let Some(c) = reader.peek_char() {
53            match c {
54                '*' => {
55                    reader.consume_char();
56
57                    // Collapse runs of `*` into one - they are equivalent.
58                    if !matches!(tokens.last(), Some(Token::AnyStr)) {
59                        tokens.push(Token::AnyStr);
60                    }
61                },
62                '?' => {
63                    reader.consume_char();
64                    tokens.push(Token::AnyChar);
65                },
66                '\\' => {
67                    reader.consume_char();
68                    match reader.consume_char() {
69                        Some(escaped) => push_char(&mut tokens, escaped),
70                        None => {
71                            return Err(Error {
72                                message: String::from("Trailing backslash in pattern"),
73                            });
74                        },
75                    }
76                },
77                '[' => {
78                    reader.consume_char();
79                    let mut variants = Vec::new();
80
81                    loop {
82                        match reader.peek_char() {
83                            Some(']') | None => break,
84                            Some(c1) => {
85                                reader.consume_char();
86                                if reader.peek_char() == Some('-') {
87                                    let saved_idx = 1;
88                                    reader.consume_char();
89                                    match reader.peek_char() {
90                                        Some(']') | None => {
91                                            variants.push(Enclosed::Char(c1));
92                                            variants.push(Enclosed::Char('-'));
93                                        },
94                                        Some(c2) => {
95                                            reader.consume_char();
96                                            variants.push(Enclosed::Range { from: c1, to: c2 });
97                                        },
98                                    }
99                                    let _ = saved_idx;
100                                } else {
101                                    variants.push(Enclosed::Char(c1));
102                                }
103                            },
104                        }
105                    }
106
107                    match reader.consume_char() {
108                        Some(']') => {},
109                        _ => {
110                            return Err(Error {
111                                message: String::from("Unterminated character class"),
112                            });
113                        },
114                    }
115
116                    tokens.push(Token::AnyOf(variants));
117                },
118                _ => {
119                    reader.consume_char();
120                    push_char(&mut tokens, c);
121                },
122            }
123        }
124
125        Ok(Self { tokens })
126    }

128    /// True iff the entire input matches the pattern.
129    pub fn matches(&self, input: &str) -> bool {
130        match_lengths(&self.tokens, input)
131            .into_iter()
132            .any(|n| n == input.len())
133    }

135    /// Returns `input` with the longest (greedy) or shortest (lazy) prefix
136    /// that matches the pattern removed. If no prefix matches the pattern,
137    /// `input` is returned unchanged.
138    pub fn trim_start<'a>(&self, input: &'a str, greedy: bool) -> &'a str {
139        let lengths = match_lengths(&self.tokens, input);
140        pick(&lengths, greedy).map_or(input, |n| &input[n..])
141    }
142
143    /// Returns `input` with the longest (greedy) or shortest (lazy) suffix
144    /// that matches the pattern removed.
145    pub fn trim_end<'a>(&self, input: &'a str, greedy: bool) -> &'a str {
146        // Walk every UTF-8-safe suffix-start and ask: does the pattern match
147        // the entire suffix? Pick the longest (greedy) or shortest (lazy).
148        let mut candidates: Vec<usize> = Vec::new();
149        let mut i = 0;
150
151        while i <= input.len() {
152            if input.is_char_boundary(i) {
153                let suffix_len = input.len() - i;
154
155                if match_lengths(&self.tokens, &input[i..])
156                    .into_iter()
157                    .any(|n| n == suffix_len)
158                {
159                    candidates.push(suffix_len);
160                }
161            }
162
163            i += 1;
164        }
165
166        pick(&candidates, greedy).map_or(input, |n| &input[..input.len() - n])
167    }

169    /// Replace the first or all occurrences of any substring matching the
170    /// pattern with `replacement`. A "match" is the longest substring at the
171    /// current scan position that the pattern matches.
172    pub fn replace(&self, input: &str, replacement: &str, all_occurrences: bool) -> String {
173        let mut out = String::with_capacity(input.len());
174        let mut i = 0;
175
176        while i < input.len() {
177            if !input.is_char_boundary(i) {
178                i += 1;
179                continue;
180            }
181
182            let lengths = match_lengths(&self.tokens, &input[i..]);
183
184            if let Some(&n) = lengths.iter().max() {
185                if n == 0 {
186                    let next = input[i..].chars().next().map_or(1, char::len_utf8);
187                    out.push_str(&input[i..i + next]);
188                    i += next;
189                    continue;
190                }
191
192                out.push_str(replacement);
193
194                i += n;
195
196                if !all_occurrences {
197                    out.push_str(&input[i..]);
198                    return out;
199                }
200            } else {
201                let next = input[i..].chars().next().map_or(1, char::len_utf8);
202                out.push_str(&input[i..i + next]);
203                i += next;
204            }
205        }
206
207        out
208    }
209}

211fn pick(lengths: &[usize], greedy: bool) -> Option<usize> {
212    if greedy {
213        lengths.iter().copied().max()
214    } else {
215        lengths.iter().copied().min()
216    }
217}

219/// Return the set of prefix byte-lengths of `input` that the token sequence
220/// fully consumes. An empty result means no match.
221fn match_lengths(tokens: &[Token], input: &str) -> Vec<usize> {
222    if tokens.is_empty() {
223        return vec![0];
224    }
225
226    let mut results: Vec<usize> = Vec::new();
227    match &tokens[0] {
228        Token::AnyStr => {
229            let mut i = 0;
230
231            loop {
232                if input.is_char_boundary(i) {
233                    for j in match_lengths(&tokens[1..], &input[i..]) {
234                        results.push(i + j);
235                    }
236                }
237
238                if i >= input.len() {
239                    break;
240                }
241
242                i += 1;
243            }
244        },
245        Token::AnyChar => {
246            if let Some(c) = input.chars().next() {
247                let n = c.len_utf8();
248
249                for j in match_lengths(&tokens[1..], &input[n..]) {
250                    results.push(n + j);
251                }
252            }
253        },
254        Token::ExactStr(s) => {
255            if input.starts_with(s.as_str()) {
256                let n = s.len();
257
258                for j in match_lengths(&tokens[1..], &input[n..]) {
259                    results.push(n + j);
260                }
261            }
262        },
263        Token::AnyOf(variants) => {
264            if let Some(c) = input.chars().next() {
265                let matches = variants.iter().any(|v| match v {
266                    Enclosed::Char(x) => *x == c,
267                    Enclosed::Range { from, to } => *from <= c && c <= *to,
268                });
269
270                if matches {
271                    let n = c.len_utf8();
272
273                    for j in match_lengths(&tokens[1..], &input[n..]) {
274                        results.push(n + j);
275                    }
276                }
277            }
278        },
279    }
280
281    results
282}
